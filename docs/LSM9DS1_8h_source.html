415  // initGyro() -- Sets up the gyroscope to begin reading.
416  // This function steps through all five gyroscope control registers.
417  // Upon exit, the following parameters will be set:
418  // - CTRL_REG1_G = 0x0F: Normal operation mode, all axes enabled.
419  // 95 Hz ODR, 12.5 Hz cutoff frequency.
420  // - CTRL_REG2_G = 0x00: HPF set to normal mode, cutoff frequency
421  // set to 7.2 Hz (depends on ODR).
422  // - CTRL_REG3_G = 0x88: Interrupt enabled on INT_G (set to push-pull and
423  // active high). Data-ready output enabled on DRDY_G.
424  // - CTRL_REG4_G = 0x00: Continuous update mode. Data LSB stored in lower
425  // address. Scale set to 245 DPS.
426  // - CTRL_REG5_G = 0x00: FIFO disabled. HPF disabled.
427  void initGyro();
428 
429  // readGyro() -- Read the gyroscope output registers.

431  // The readings are stored in the class' gx, gy, and gz variables. Read
432  // those _after_ calling readGyro().
433  void readGyro();
434 
435  // initAccel() -- Sets up the accelerometer to begin reading.
436  // This function steps through all accelerometer related control registers.
437  // Upon exit these registers will be set as:
438  // - CTRL_REG0_XM = 0x00: FIFO disabled. HPF bypassed. Normal mode.
439  // - CTRL_REG1_XM = 0x57: 100 Hz data rate. Continuous update.
440  // all axes enabled.
441  // - CTRL_REG2_XM = 0x00: 2g scale. 773 Hz anti-alias filter BW.
442  // - CTRL_REG3_XM = 0x04: Accel data ready signal on INT1_XM pin.
443  void initAccel();

451  // initMag() -- Sets up the magnetometer to begin reading.
452  // This function steps through all magnetometer-related control registers.
453  // Upon exit these registers will be set as:
454  // - CTRL_REG4_XM = 0x04: Mag data ready signal on INT2_XM pin.
455  // - CTRL_REG5_XM = 0x14: 100 Hz update rate. Low resolution. Interrupt
456  // requests don't latch. Temperature sensor disabled.
457  // - CTRL_REG6_XM = 0x00: 2 Gs scale.
458  // - CTRL_REG7_XM = 0x00: Continuous conversion mode. Normal HPF mode.
459  // - INT_CTRL_REG_M = 0x09: Interrupt active-high. Enable interrupts.
